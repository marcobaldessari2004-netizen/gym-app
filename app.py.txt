import streamlit as st
import google.generativeai as genai
import os

# Titolo della tua App
st.title("La mia AI Personale ðŸ¤–")

# Configurazione (La chiave la metteremo dopo nelle impostazioni sicure)
# Se stai provando in locale puoi scommentare la riga sotto, ma per l'online usa i "Secrets"
api_key = st.secrets["GEMINI_API_KEY"] 
genai.configure(api_key=api_key)

# Scegli il modello (lo stesso che usavi in AI Studio, es. gemini-pro o 1.5-flash)
model = genai.GenerativeModel('gemini-1.5-flash')

# Inizializza la chat se non esiste
if "messages" not in st.session_state:
    st.session_state.messages = []

# Mostra i messaggi precedenti
for message in st.session_state.messages:
    with st.chat_message(message["role"]):
        st.write(message["content"])

# Casella di input per l'utente
if prompt := st.chat_input("Scrivi qui..."):
    # Mostra il messaggio dell'utente
    st.session_state.messages.append({"role": "user", "content": prompt})
    with st.chat_message("user"):
        st.write(prompt)

    # Risposta dell'AI
    with st.chat_message("assistant"):
        try:
            # Prepara la storia della chat per il modello
            chat_history = [
                {"role": m["role"], "parts": [m["content"]]} 
                for m in st.session_state.messages
                if m["role"] != "system" # Filtra eventuali messaggi di sistema
            ]

            # Chiedi a Gemini
            response = model.generate_content(chat_history)
            st.write(response.text)

            # Salva la risposta
            st.session_state.messages.append({"role": "model", "content": response.text})
        except Exception as e:
            st.error(f"Errore: {e}")
